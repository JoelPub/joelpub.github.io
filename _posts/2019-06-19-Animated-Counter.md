<a href="https://codepen.io/anon/pen/Wqoemz">Animated Counter on Scroll</a>


<pre><code>

<div id="counter">
    <div class="counter-value" data-count="300">0</div>
</div>

      $({
        countNum: 0
      }).animate({
          countNum: 30
        },

        {

          duration: 2000,
          easing: 'swing',
          step: function() {
            $('.counter-value').text(Math.floor(this.countNum));
          },
          complete: function() {
            $('.counter-value').text(this.countNum);
            //alert('finished');
          }

        });

</code></pre>
